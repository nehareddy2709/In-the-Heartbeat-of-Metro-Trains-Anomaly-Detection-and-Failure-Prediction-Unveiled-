import matplotlib.pyplot as plt

# Assuming 'data' contains your dataset with columns including 'compressor_failure' and sensors
# Filter data for compressor failure and success
failure_data = data[data['COMP'] == 1]
success_data = data[data['COMP'] == 0]

# Count occurrences of each category for each sensor
failure_counts = failure_data[['DV_eletric', 'Towers', 'MPG', 'LPS', 'Pressure_switch', 'Oil_level', 'Caudal_impulses']].sum()
success_counts = success_data[['DV_eletric', 'Towers', 'MPG', 'LPS', 'Pressure_switch', 'Oil_level', 'Caudal_impulses']].sum()

# Plot pie chart
labels = ['DV_eletric', 'Towers', 'MPG', 'LPS', 'Pressure_switch', 'Oil_level', 'Caudal_impulses']
fig, ax = plt.subplots(1, 2, figsize=(12, 6))
ax[0].pie(failure_counts, labels=labels, autopct='%1.1f%%')
ax[0].set_title('Compressor Failure')
ax[1].pie(success_counts, labels=labels, autopct='%1.1f%%')
ax[1].set_title('Compressor Success')
plt.show()

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.model_selection import cross_val_score
from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import accuracy_score


# Define features (X) and target variable (y)
# X = data[['DV_eletric', 'Towers', 'MPG', 'LPS', 'Pressure_switch', 'Oil_level', 'Caudal_impulses']]
X = data[['Towers', 'MPG', 'Caudal_impulses']]
y = data['COMP']

# Convert the target variable into classes (binary classification)
threshold = 0.5  # Define the threshold
y_class = (y > threshold).astype(int)

# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y_class, test_size=0.3, random_state=42)

# Select model (Decision Tree Classifier)
model = DecisionTreeClassifier(max_depth=5, min_samples_split=10, min_samples_leaf=5)

# Train the model

model.fit(X_train, y_train)

# Make predictions
y_pred = model.predict(X_test)

# Evaluate the model
accuracy = accuracy_score(y_test, y_pred)
print("Accuracy:", accuracy)

# Perform cross-validation
cv_scores = cross_val_score(model, X_train, y_train, cv=5)

# Print cross-validation scores and mean accuracy
print("Cross-validation Scores:", cv_scores)
print("Mean CV Accuracy:", cv_scores.mean())
